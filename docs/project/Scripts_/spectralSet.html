{

max_buff--;

{
if( c == '+')
c = ' ';
else if( c == '%')
{
int i, c1;
c = 0;
for( i = 0; i < 2; i++)
{
c *= 16;
c1 = *tptr++;

c += c1 - '0';
else if( c1 >= 'A' && c1 <= 'F')
c += c1 - 'A' + 10; /* wasn't an hex digit: */
else
return( -1); /* not supposed to happen */
}
}

}
